'''Given a 2D matrix M of dimensions RxC. Find the maximum sum submatrix in it.'''

def kadane(arr):
    curr = arr[0]
    best = arr[0]

    for i in range(1, len(arr)):
        curr = max(arr[i], curr + arr[i])
        best = max(best, curr)

    return best


r = int(input("Rows: "))
c = int(input("Columns: "))

matrix = []

print("Enter matrix:")

for i in range(r):
    matrix.append(list(map(int, input().split())))

result = float('-inf')

for left in range(c):

    temp = [0] * r

    for right in range(left, c):

        for i in range(r):
            temp[i] += matrix[i][right]

        result = max(result, kadane(temp))

print(result)
