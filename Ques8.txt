'''Given a Binary Tree (BT), convert it to a Doubly Linked List(DLL) In-Place. The left and right
pointers in nodes are to be used as previous and next pointers respectively in converted DLL. The
order of nodes in DLL must be same as Inorder of the given Binary Tree. The first node of Inorder
traversal (leftmost node in BT) must be the head node of the DLL.'''


class Node:
    def __init__(self, data):
        self.data = data
        self.left = None
        self.right = None


root = Node(int(input("Root: ")))
root.left = Node(int(input("Left child: ")))
root.right = Node(int(input("Right child: ")))

prev = None
head = None


def inorder(node):
    global prev, head

    if not node:
        return

    inorder(node.left)

    if prev is None:
        head = node
    else:
        node.left = prev
        prev.right = node

    prev = node

    inorder(node.right)


inorder(root)

curr = head

while curr:
    print(curr.data, end=" ")
    curr = curr.right
